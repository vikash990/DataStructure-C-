#include<bits/stdc++.h>
using namespace std;

#define MAX 1000

class Stack {

  int top;

public:
  int a[MAX];
  Stack() {top = -1;}
  bool push(int x);
  int pop();
  int peek();
  bool isEmpty();
};

bool Stack::push(int x)
{

  if(top>=(MAX-1)){
   cout<<"Stack OverFlow";
   return false;
}
 else {

  a[++top]=x;
  cout<<x<<"Element Pushed";
   return true;
}

}

int Stack::pop()
{
 if(top<0){
 cout<<"Stack UnderFlow";
  return 0;


} else {
 int x=a[top--];
 return x;

}


}

int Stack::peek()
{
 
 if(top<0){
 cout<<"Stack UnderFlow";
  return 0;


} else

{ int x=a[top];
   return x;

}

}


bool Stack::isEmpty()
{
  return(top<0);

}

 class Stack s;

   cout<<"Peek Element"<<s.peek();
   cout<<"\n";
  cout<<"Element popped"<<s.pop();

return 0;
